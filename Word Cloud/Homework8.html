##read tweets text file,split data and add custom words to STOPWORDS
with open('TwitterData.txt', encoding="utf8") as file:
    words = file.read()
        
wordsData = " ".join([word for word in words.split()
                            if 'http' not in word
                                and not word.startswith('@')
                                and word != 'RT'
                            ])
                                   
STOPWORDS.add('amp')    
STOPWORDS.add('want')
STOPWORDS.add('new')
STOPWORDS.add('via')
STOPWORDS.add('man')
STOPWORDS.add('will')
STOPWORDS.add('here')
STOPWORDS.add('Heres')
STOPWORDS.add('Here')
STOPWORDS.add('https')
STOPWORDS.add('one')
STOPWORDS.add('ve')
